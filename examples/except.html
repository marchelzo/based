import os (..)
import errno

function openFile(path) {
    match open(path, [O_RDONLY]) {
        -1 => { throw Err(errno::get()); },
        fd => { return fd; }
    }
}

try {
    let fd = openFile('does_not_exist.txt');
    print(fd);
} catch Err(e) {
    print("Failed to open file: {errno::str(e)}");
}
